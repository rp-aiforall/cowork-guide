const DEPT_LABELS = {core:'Core',fin:'Finance',sales:'Sales',eng:'Product & Eng',hr:'HR / People',mkt:'Marketing',legal:'Legal',cs:'Customer Success'};

let currentDept = 'core';

let visited = new Set(JSON.parse(localStorage.getItem('rp2_visited')||'[]'));


function deptSections(d){ return ALL_SECTIONS.filter(s=>s.id.startsWith(d==='core'?'c-':d==='fin'?'f-':d==='sales'?'s-':d==='eng'?'e-':d==='hr'?'h-':d==='mkt'?'m-':d==='legal'?'l-':'x-')) }


function buildDeptBar(){

  const bar=document.getElementById('dept-bar');

  bar.innerHTML='';

  DEPTS.forEach(d=>{

    const btn=document.createElement('button');

    btn.className='dept-btn '+(d.cls||'')+(d.id===currentDept?' active':'');

    btn.textContent=d.label;

    btn.onclick=()=>switchDept(d.id);

    bar.appendChild(btn);

  });

}


function buildNav(){

  const nav=document.getElementById('nav');

  nav.innerHTML='';

  const secs=deptSections(currentDept);

  let lastGroup='';

  secs.forEach(s=>{

    if(s.group!==lastGroup){

      if(lastGroup) nav.insertAdjacentHTML('beforeend','<div class="nav-divider"></div>');

      nav.insertAdjacentHTML('beforeend',`<div class="nav-group">${s.group}</div>`);

      lastGroup=s.group;

    }

    const v=visited.has(s.id)?'visited':'';

    nav.insertAdjacentHTML('beforeend',`

      <button class="nav-item ${v}" id="nav-${s.id}" onclick="showSection('${s.id}')">

        <span class="nav-num">${s.num}</span>

        <span style="flex:1;text-align:left;line-height:1.3">${s.title}</span>

        <span class="check"><svg width="13" height="13" viewBox="0 0 13 13"><circle cx="6.5" cy="6.5" r="6.5" fill="#378ADD" opacity=".15"/><path d="M3.5 6.5l2 2 4-4" stroke="#185FA5" stroke-width="1.4" stroke-linecap="round" stroke-linejoin="round" fill="none"/></svg></span>

      </button>`);

  });

}


function updateProgress(){

  const all=ALL_SECTIONS.length;

  const v=visited.size;

  const pct=Math.round(v/all*100);

  document.getElementById('pbar').style.width=pct+'%';

  document.getElementById('plabel').textContent=`${v} of ${all} sections visited`;

}


function switchDept(deptId){

  currentDept=deptId;

  buildDeptBar();

  buildNav();

  const first=deptSections(deptId)[0];

  if(first) showSection(first.id);

}


function showSection(id){

  const s=ALL_SECTIONS.find(x=>x.id===id);

  if(!s) return;

  document.querySelectorAll('.nav-item').forEach(el=>el.classList.remove('active'));

  const navEl=document.getElementById('nav-'+id);

  if(navEl){navEl.classList.add('active');navEl.classList.add('visited');}

  visited.add(id);

  localStorage.setItem('rp2_visited',JSON.stringify([...visited]));

  updateProgress();

  const d=DEPTS.find(x=>x.id===s.dept)||DEPTS[0];

  const prompt=encodeURIComponent(s.prompt);

  document.getElementById('section-content').innerHTML=`

    <div class="section-badge ${d.badgeCls}">${s.num} — ${DEPT_LABELS[s.dept]||s.dept}</div>
